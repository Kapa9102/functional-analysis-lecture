\begin{corollary}[]
Let $G $ be a vector space, and let $B $ be a subset of $G $. Assume that:
\[
\forall f \in   G': \quad 
f(B)  = \left\{ f(b) :  b \in   B \right\} \text{ is bounded }   \implies 
\text{ $B $ is bounded. }  
\]
\end{corollary}
\begin{proof}
Consider: 
\[
\begin{array}{cccc}
      T_{b} : &  G'  & \longrightarrow & \RR  \\

           &  f  & \longmapsto     & T_{b}(f) = 
           \left\langle f, b \right\rangle = f(b) \\ 
\end{array}
\]
Then using (B.S) principle, we get 
\[
\left| \left\langle f, b \right\rangle  \right|  \leq 
c  \| f \|.
\]
Thus we get
\[
\| b \| \leq c, \quad \forall b \in   B,
\]
which means that $B $ is bounded.
\end{proof}

Let $G $ be a banach space and let $B' $ be a subset of $G' $ assume that:
\[
\forall x \in   G: \quad 
\left\langle B', x \right\rangle  = 
\left\{ \left\langle f,x \right\rangle : f \in   G' \right\} \text{ is bounded }  
\implies 
B' \text{ is bounded. }  
\]
\end{corollary}
\begin{theorem}[Open mapping theorem]
  Let $E $, and $F $ be two Banach spaces, and let $T \in  \mathcal{L} (E, F)  $ assume that $T $ is onto, 
  then there exists $c > 0 $, such that:
  \[
  B_{F}(0, c) \subset T(B_{E}(0, 1) )  \quad \quad (*) 
  \]
\end{theorem}
$(*)$ can be translated into image of open set is open. 
\begin{corollary}
  Let $(E, \| . \| _{1})  $, and $(E, \| . \| _{2})  $ be Banach spaces. If there exists a constant 
  $c > 0 $, such that: 
  \[
  \| x \| _{1} \leq  c \| x \| _{2} \quad (\forall  x \in   E) 
  \]
  then  $\| . \| _{1} $, and $\| . \| _{2} $ are equivalents.
\end{corollary}
\begin{corollary}[]
Let $E $, and $F $ be two Banach spaces, and let $T \in  \mathcal{L} (E, F) $ assume that 
$T $ is bijective, then 
\[
T^{-1} \in   \mathcal{L} (F, E) 
\]
\end{corollary}
